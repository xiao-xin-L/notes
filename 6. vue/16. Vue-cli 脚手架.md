## Vue-cli 脚手架

1. 自己写一个工程化开发环境

   ```js
   // 新增知识点
   // vue-router(vue路由) vuex(vue的全局管理器) vue-loader(打包vue文件的工具)
   npm install vue vue-router vuex vue-loader
   // webpack.common.config.js的配置信息
   // 引入vue-loader的插件文件
   let VueLoaderPlugin = require("vue-loader/lib/plugin.js");
   module.exports = {
       module: {
           rules: [
               {
                   test: /\.vue$/,
                   loader: "vue-loader"
               }
           ]
       },
       plugins: [
           new VueLoaderPlugin()
       ]
   }
   // 当后缀为ico的文件打包不进dist文件时
   // 可以单独配置信息
   module.exports = {
       module: {
           rules: [
               {
                   test: /\.ico$/,
                   use: {
                       loader: "file-loader",
                       options: {
                           name: "[name].[ext]",
                           outputPath: "./",
                           esModule: false
                       }
                   }
               }
           ]
       }
   }
   // 在index.js文件引入
   import "./favicon.ico"
   
   // 当vue文件中template有图片引入时, 需要在options中设置esModule: false, 否则显示不出来
   
   // html-withimg-plugin 解决直接在HTML中使用img时没有将图片文件打包的问题
   
   // 默认提示是没有写入后缀的
   // 可以通过设置以下代码进行配置
   resolve: {
   	extensions: [
   		".js", ".json", ".vue", ".scss", ".css"
   	]
   }
   // App.vue文件
   <template>
       // 写HTML标签的
       <div>{{ info }}</div>
   </template>
   <script>
       import HelloWorld from "./components/HelloWorld.vue"
       exports default {
   		name: "App",
       	components: {
               HelloWorld
           },
           data(){
               return {
                   info: "hello world"
               }
           }
   	}
   </script>
   <style>
       // 写css样式的
       div {
           background-color: blue;
       }
   </style>
   ```

2. Vue-cli 脚手架

   ```js
   // Vue-cli 的安装
   npm install @vue/cli
   // 项目的创建
   vue create "项目名称"
   // 创建后package.json中的命令
   serve: 开发环境
   build: 生产环境
   lint: eslint
   // 由于vue-cli把webpack的文件给隐藏了, 所以我们想进行一些精细化配置, 可以创建vue.config.js来配置文件
   // 这个文件使用的vue的一些接口, 不能使用webpack的原生配置
   module.exports = {
       publicPath: "/", // 资源的根路径
       outputDir: "./dist", // 运行生产环境时构建文件的目录
       assetDir: "", // 放置生成的静态资源 (js、css、img、fonts) 的 (相对于 outputDir 的) 目录
       indexPath: "index.html", // 指定生成的 index.html 的输出路径 (相对于 outputDir)
       filenameHashing: false, // 生成的静态资源后面是否生成加上哈希值
       pages: {
           index: {
               // page 的入口
               entry: "./src/index.js",
               // 模板来源
               template: "./public/index.html",
               // 打包后的文件名
               filename: "main.html",
               // 当使用 title 选项时
               // template 中的 title 标签应该是 <titile><%= htmlWebpackPlugin.options.title %></title>
               title: "Index Page",
               chunks: []
           }
       },
       css: {
           extract: false, // 是否把组件中的css提取出来单独放在一个css文件中
           sourceMap: false, // 是否为css开启 source map
           requireModuleExtension: false, // 去除文件名的.module
           loaderOptions: {
               // 每个选项都会传递给相应的loader
               css: {},
               postcss: {},
               sass: {},
               less: {},
               stylus: {}
           }
       },
       devServer: {
           proxy: "https://localhost:8080"
       },
       // 这里的配置和webpack.config.js完全一样
       configureWebpack: {
           entry: "",
           output: {},
           module: {},
           plugins: []
       },
       chainWebpack: config => {
           
       }
   }
   ```
